            this.map.data.setStyle(function(feature) {

                const lon = feature.getProperty('PREVALENCE');

                const value = 255 - Math.round(mapNumber(lon, 0, 5, 0, 255));
                const color = 'rgb(' + value + ',' + value + ',' + 0 + ')';

                return {
                    fillColor: color,
                    strokeWeight: 1
                };
